# Qt requires moc-ing
set(GetSetGui_QOBJ_H GetSetWidget.h GetSetSettingsWindow.h )
QT4_WRAP_CPP(GetSetGui_MOC ${GetSetGui_QOBJ_H})

# build library
add_library(GetSetGui STATIC
	GetSetWidget.h
	GetSetWidget.cpp
	GetSetSettingsWindow.h
	GetSetSettingsWindow.cpp
	${GetSetGui_MOC}
)

# GetSetGui depends on GetSet
add_dependencies(GetSetGui GetSet)

# set version of the library
set_target_properties(GetSetGui PROPERTIES VERSION ${GETSET_VERSION})

# Define headers to install
set_target_properties(GetSetGui PROPERTIES PUBLIC_HEADER "GetSetWidget.h;GetSetSettingsWindow.h")

# define install destinations
#install(TARGETS GetSetGui RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE DESTINATION lib PUBLIC_HEADER DESTINATION ${INCLUDE_INSTALL_DIR}/GetSetGui)
install(TARGETS GetSetGui EXPORT GetSetExport LIBRARY DESTINATION ${LIB_INSTALL_DIR} ARCHIVE DESTINATION ${LIB_INSTALL_DIR} RUNTIME DESTINATION ${BIN_INSTALL_DIR} PUBLIC_HEADER DESTINATION ${INCLUDE_INSTALL_DIR}/GetSetGui )
